    def self.search(query = '', window = :auto, page = 1, perpage = 10)
      if [:auto, :h, :d, :w, :m, :a].include? window
        resp = get_response("/search.json?q=#{query}&window=#{window}&page=#{page}&perpage=#{perpage}")
        Page.new(resp, :link_search_result)
      else
        nil
      end      
    end
